FIND_PACKAGE(PkgConfig REQUIRED)

PKG_CHECK_MODULES(PIXMAN REQUIRED pixman-1)
INCLUDE_DIRECTORIES(${PIXMAN_INCLUDE_DIRS})
LINK_DIRECTORIES(${PIXMAN_LIBRARY_DIRS})

PKG_CHECK_MODULES(JSON REQUIRED json-c)
INCLUDE_DIRECTORIES(${JSON_INCLUDE_DIRS})
LINK_DIRECTORIES(${JSON_LIBRARY_DIRS})

option(WITH_TCMALLOC "Build with TCMALLOC" ON)

include_directories(
	${CMAKE_SOURCE_DIR}/asst
	${CMAKE_SOURCE_DIR}/kits
	${CMAKE_SOURCE_DIR}/helper
	${CMAKE_SOURCE_DIR}/tool
	${CMAKE_BINARY_DIR}/tool
	${CMAKE_SOURCE_DIR}/db
	${CMAKE_SOURCE_DIR}/play
	${CMAKE_SOURCE_DIR}/cook
	${CMAKE_SOURCE_DIR}/action
	)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS}")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${CMAKE_C_FLAGS} -std=c++11")
set(CMAKE_LINK_FLAGS "${CMAKE_LINK_FLAGS}")

add_executable(nemoart
	nemoart.c
	)

TARGET_LINK_LIBRARIES(nemoart
	nemoasst
	nemokits
	nemotool
	nemoplay
	nemocook
	nemoaction
	)

if(WITH_TCMALLOC)
	TARGET_LINK_LIBRARIES(nemoart
		tcmalloc_minimal
		)
endif(WITH_TCMALLOC)

SET_TARGET_PROPERTIES(nemoart PROPERTIES OUTPUT_NAME "exec")

INSTALL(TARGETS nemoart DESTINATION ${CMAKE_INSTALL_PACKAGES}/nemo.art)
INSTALL(FILES ${CMAKE_CURRENT_SOURCE_DIR}/manifest.json DESTINATION ${CMAKE_INSTALL_PACKAGES}/nemo.art/)
