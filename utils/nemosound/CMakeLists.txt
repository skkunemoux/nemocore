FIND_PACKAGE(PkgConfig REQUIRED)

PKG_CHECK_MODULES(GLIB REQUIRED glib-2.0)
INCLUDE_DIRECTORIES(${GLIB_INCLUDE_DIRS})
LINK_DIRECTORIES(${GLIB_LIBRARY_DIRS})

PKG_CHECK_MODULES(LIBPULSE REQUIRED libpulse)
INCLUDE_DIRECTORIES(${LIBPULSE_INCLUDE_DIRS})
LINK_DIRECTORIES(${LIBPULSE_LIBRARY_DIRS})

PKG_CHECK_MODULES(GLIBPULSE REQUIRED libpulse-mainloop-glib)
INCLUDE_DIRECTORIES(${GLIBPULSE_INCLUDE_DIRS})
LINK_DIRECTORIES(${GLIBPULSE_LIBRARY_DIRS})

include_directories(
	${CMAKE_SOURCE_DIR}/asst
	${CMAKE_SOURCE_DIR}/helper
	${CMAKE_SOURCE_DIR}/tool
	${CMAKE_BINARY_DIR}/tool
	)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS}")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${CMAKE_C_FLAGS}")
set(CMAKE_LINK_FLAGS "${CMAKE_LINK_FLAGS}")

add_executable(nemosound
	nemosound.c
	nemoglib.c
	)

TARGET_LINK_LIBRARIES(nemosound
	nemoasst
	nemohelper
	nemotool
	${GLIB_LIBRARIES}
	${LIBPULSE_LIBRARIES}
	${GLIBPULSE_LIBRARIES}
	)

INSTALL(TARGETS nemosound DESTINATION bin)

add_executable(nemopulse
	nemopulse.c
	)

TARGET_LINK_LIBRARIES(nemopulse
	nemoasst
	nemohelper
	${LIBPULSE_LIBRARIES}
	)

INSTALL(TARGETS nemopulse DESTINATION bin)
